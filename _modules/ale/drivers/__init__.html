[docs]def load(label):
    """
    Attempt to load a given label from all possible drivers

    Parameters
    ----------
    label : str
               String path to the given label file
    """
    for name, driver in drivers.items():
        try:
            print("TRYING:", driver)
            res = driver(label)
            if res.is_valid():
                with res as r:
                    return res

        except Exception as e:
            import traceback
            traceback.print_exc()
    raise Exception('No Such Driver for Label')
